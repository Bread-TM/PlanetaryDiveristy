# In this mod, the vanilla planet classes will no longer use the wet, dry and cold climates, since most habitable planet classes may have more than one climate type
# So for compatibility with deposit spawning and vanilla scripts, we must overwrite the climate triggers to something else other than has_climate = XX

is_wet = {
	has_climate = wet
}

is_dry = {
	has_climate = dry
	NOT = { is_planet_class = pc_city }
}

is_barren = {
	OR = {
		is_planet_class = pc_barren
		is_planet_class = pc_barren_cold
		is_planet_class = pc_asteroid
	}
}

is_cold = {
	has_climate = cold
}

ideal_planet_is_wet = {
	OR = {
		ideal_planet_class = pc_continental
		ideal_planet_class = pc_ocean
		ideal_planet_class = pc_tropical
	}
}

ideal_planet_is_dry = {
	OR = {
		ideal_planet_class = pc_desert
		ideal_planet_class = pc_arid
		ideal_planet_class = pc_savannah
	}
}

ideal_planet_is_cold = {
	OR = {
		ideal_planet_class = pc_tundra
		ideal_planet_class = pc_arctic
		ideal_planet_class = pc_alpine
	}
}
