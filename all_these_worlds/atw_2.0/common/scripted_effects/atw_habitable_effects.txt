# SUPER-TERRESTRIAL CLASSES

roll_chloerosgaean = {
	# Remove previously spawned modifiers
	remove_atmosphere_modifier = yes
	remove_class_modifier = yes
	# Spawn atomosphere modifier
	add_modifier = { modifier = "atmosphere-thick_blue" days = -1 }
	# Spawn planet class modifier and update texture/portrait/atmosphere color and intensity	
	add_modifier = { modifier = "biosphere-seeded" days = -1 }
	add_modifier = { modifier = "class_chloerosgaean" days = -1 }
	set_planet_flag = "habitable_converted"
	random_list = {
		0 = {
			modifier = { add = 10 is_colony = yes } # Colonized worlds will use biome portrait
			set_planet_entity = {
				picture = "biome_chloerosgaean"
				entity = "super_terrestrial_chloerosgaean_01_entity"
				atmosphere_color 		= hsv { 0.59 0.45 0.95 }
				atmosphere_intensity 	= 1.0
				atmosphere_width 		= 0.5
			}
		}
		0 = {
			modifier = { add = 10 is_colony = no } # Uncolonized worlds will use orbit portrait
			set_planet_entity = {
				picture = "orbit_chloerosgaean"
				entity = "super_terrestrial_chloerosgaean_01_entity"
				atmosphere_color 		= hsv { 0.59 0.45 0.95 }
				atmosphere_intensity 	= 1.0
				atmosphere_width 		= 0.5
			}
		}
	}
}

# TERRESTRIAL CLASSES

roll_chloerosorimos = {
	# Remove previously spawned modifiers
	remove_atmosphere_modifier = yes
	remove_class_modifier = yes
	# Spawn atomosphere modifier
	add_modifier = { modifier = "atmosphere-thick_blue" days = -1 }
	
	# Homeworlds must roll the default biosphere
	if = {
		limit = { is_homeworld = yes }
		add_modifier = { modifier = "biosphere-diverse" days = -1 }
		# Spawn planet class modifier and update texture/portrait/atmosphere color and intensity
		add_modifier = { modifier = "class_chloerosorimos" days = -1 }
		set_planet_entity = {
			picture = "biome_chloerosorimos_diverse"
			entity = "terrestrial_chloerosorimos_diverse_01_entity"
			atmosphere_color 		= hsv { 0.59 0.45 0.95 }
			atmosphere_intensity 	= 1.0
			atmosphere_width 		= 0.5
		}
		set_planet_flag = "habitable_converted"
	}
	# If not, randomize the biosphere with the proper texture
	else = {
		random_list = {
			1 = {
				# Roll with Dense biosphere
				add_modifier = { modifier = "biosphere-dense" days = -1 }
				# Spawn planet class modifier and update texture/portrait/atmosphere color and intensity
				add_modifier = { modifier = "class_chloerosorimos" days = -1 }
				random_list = {
					0 = {
						modifier = { add = 10 is_colony = yes } # Colonized worlds will use biome portrait
						set_planet_entity = {
							picture = "biome_chloerosorimos_dense"
							entity = "terrestrial_chloerosorimos_dense_01_entity"
							atmosphere_color 		= hsv { 0.59 0.45 0.95 }
							atmosphere_intensity 	= 1.0
							atmosphere_width 		= 0.5
						}
					}
					0 = {
						modifier = { add = 10 is_colony = no } # Uncolonized worlds will use orbit portrait
						set_planet_entity = {
							picture = "orbit_chloerosorimos"
							entity = "terrestrial_chloerosorimos_dense_01_entity"
							atmosphere_color 		= hsv { 0.59 0.45 0.95 }
							atmosphere_intensity 	= 1.0
							atmosphere_width 		= 0.5
						}
					}
				}
				set_planet_flag = "habitable_converted"
				# Dense biospheres favored with bright stars (but not too bright)
				modifier = { factor = 0.25 orbits_red_star = yes }
				modifier = { factor = 1.00 orbits_orange_star = yes }
				modifier = { factor = 5.00 orbits_yellow_star = yes }
				modifier = { factor = 7.50 orbits_white_star = yes }
				modifier = { factor = 0.00 orbits_blue_star = yes }
				# Terraforming-related modifiers
				modifier = { factor = 0.25 has_planet_flag = "has_barren_preference" }
				modifier = { factor = 2.00 has_planet_flag = "has_sparse_preference" }
				modifier = { factor = 2.00 has_planet_flag = "has_diverse_preference" }
				modifier = { factor = 10.00 has_planet_flag = "has_dense_preference" }
			}
			1 = {
				# Roll with Diverse biosphere
				add_modifier = { modifier = "biosphere-diverse" days = -1 }
				# Spawn planet class modifier and update texture/portrait/atmosphere color and intensity
				add_modifier = { modifier = "class_chloerosorimos" days = -1 }
				random_list = {
					0 = {
						modifier = { add = 10 is_colony = yes } # Colonized worlds will use biome portrait
						set_planet_entity = {
							picture = "biome_chloerosorimos_diverse"
							entity = "terrestrial_chloerosorimos_diverse_01_entity"
							atmosphere_color 		= hsv { 0.59 0.45 0.95 }
							atmosphere_intensity 	= 1.0
							atmosphere_width 		= 0.5
						}
					}
					0 = {
						modifier = { add = 10 is_colony = no } # Uncolonized worlds will use orbit portrait
						set_planet_entity = {
							picture = "orbit_chloerosorimos"
							entity = "terrestrial_chloerosorimos_diverse_01_entity"
							atmosphere_color 		= hsv { 0.59 0.45 0.95 }
							atmosphere_intensity 	= 1.0
							atmosphere_width 		= 0.5
						}
					}
				}
				set_planet_flag = "habitable_converted"
				# Diverse biospheres favored with yellow stars
				modifier = { factor = 0.50 orbits_red_star = yes }
				modifier = { factor = 3.00 orbits_orange_star = yes }
				modifier = { factor = 10.00 orbits_yellow_star = yes }
				modifier = { factor = 3.00 orbits_white_star = yes }
				modifier = { factor = 0.00 orbits_blue_star = yes }
				# Terraforming-related modifiers
				modifier = { factor = 0.25 has_planet_flag = "has_barren_preference" }
				modifier = { factor = 2.00 has_planet_flag = "has_sparse_preference" }
				modifier = { factor = 10.00 has_planet_flag = "has_diverse_preference" }
				modifier = { factor = 2.00 has_planet_flag = "has_dense_preference" }
			}
			0 = {
				# Roll with Sparse biosphere
				add_modifier = { modifier = "biosphere-sparse" days = -1 }
				# Spawn planet class modifier and update texture/portrait/atmosphere color and intensity
				add_modifier = { modifier = "class_chloerosorimos" days = -1 }
				set_planet_flag = "habitable_converted"
				# Sparse biospheres favored with orange stars
				modifier = { factor = 5.00 orbits_red_star = yes }
				modifier = { factor = 6.00 orbits_orange_star = yes }
				modifier = { factor = 4.00 orbits_yellow_star = yes }
				modifier = { factor = 3.00 orbits_white_star = yes }
				modifier = { factor = 0.00 orbits_blue_star = yes }
			}
			0 = {
				# Roll with Barren biosphere
				add_modifier = { modifier = "biosphere-barren" days = -1 }
				# Spawn planet class modifier and update texture/portrait/atmosphere color and intensity
				add_modifier = { modifier = "class_chloerosorimos" days = -1 }
				set_planet_flag = "habitable_converted"
				# Barren biospheres favored with non ideal stars
				modifier = { factor = 6.00 orbits_red_star = yes }
				modifier = { factor = 1.00 orbits_orange_star = yes }
				modifier = { factor = 1.00 orbits_yellow_star = yes }
				modifier = { factor = 3.00 orbits_white_star = yes }
				modifier = { factor = 5.00 orbits_blue_star = yes }
			}
		}
	}	
}
