triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = { is_gestalt = no }
		has_major_upgraded_capital = yes
		OR = {
			any_moon = { has_modifier = pd_moon_base_trade }
			AND = {
				is_moon = yes
				solar_system = {
					any_system_planet = {
						has_moon = yes
						any_moon = {
							has_modifier = pd_moon_base_trade
						}
					}
				}
			}
		}
	}
	job_merchant_add = 1
	planet_housing_add = 1
	mult = solar_system.value:num_pd_moon_bases_trade
}
