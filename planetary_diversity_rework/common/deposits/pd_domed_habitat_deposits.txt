d_pd_moon_base_trade = {
	is_for_colonizable = yes
	icon = d_pd_moon_base
	category = deposit_cat_rare
	potential = { always = no }

	planet_modifier = {
		planet_max_districts_add = 4
		planet_max_buildings_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
			}
		}
		modifier = {
			job_merchant_add = 2
		}
	}

	inline_script = {
		script = jobs/pd_trader_per_pop
		AMOUNT = 0.1
	}
}
